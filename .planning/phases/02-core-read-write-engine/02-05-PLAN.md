  export async function resolveBackend(): Promise<IOBackend> {

    switch (env) {
      case 'node': return (await import('./node.js')).nodeBackend;
      case 'bun': return (await import('./bun.js')).bunBackend;
      case 'deno': return (await import('./deno.js')).denoBackend;

    }
  }

**Router** (`src/router.ts`):

`routedRead<T>(source: string | Blob | File, options?: ReadOptions): Promise<T>`:
1. **Blob/File detection:** Check if `source` is an instance of `Blob` (File extends Blob, so this catches both).
   - If Blob/File:
     a. Determine format: If `source` is a `File` (has `.name` property), call `detectFormat(source.name)`. Otherwise, use MIME type: `detectFormat('', source.type)`. If neither yields a format, throw OmniFormatError("Cannot detect format from Blob without filename or MIME type", "unknown").
     b. Get bytes: `const bytes = new Uint8Array(await source.arrayBuffer())`
     c. Get parser: `const parser = getParser(format)`
     d. Return `parser.parse(bytes, options) as T`
     e. Note: No backend needed — bytes come directly from the Blob.
2. **String path (existing flow):**
   a. `const format = detectFormat(source)` — if null, throw OmniFormatError
   b. `const backend = await resolveBackend()`
   c. `const bytes = await backend.readBytes(source)`
   d. `const parser = getParser(format)`
   e. Return `parser.parse(bytes, options) as T`

`routedWrite(path: string, data: unknown, options?: WriteOptions): Promise<void>`:
1. `const format = detectFormat(path)` — if null, throw OmniFormatError
2. `const serializer = getSerializer(format)`
3. `const output = serializer.serialize(data, options)`
4. `const bytes = typeof output === 'string' ? new TextEncoder().encode(output) : output`
5. `const backend = await resolveBackend()`
6. Use `backend.writeAtomic(path, bytes)` for server runtimes (detected via detectEnv), `backend.writeBytes(path, bytes)` for browser

`routedWriteRaw(path: string, data: Uint8Array | string): Promise<void>`:
1. `const bytes = typeof data === 'string' ? new TextEncoder().encode(data) : data`
2. `const backend = await resolveBackend()`
3. Use writeAtomic for server, writeBytes for browser

`routedStream(path: string, options?: ReadOptions): Promise<ReadableStream>`:
1. `const format = detectFormat(path)` — if null or format not streamable (only csv, tsv, text are streamable), throw OmniFormatError
2. `const backend = await resolveBackend()`
3. If !backend.readStream, throw OmniReadError("Streaming not supported in this runtime", path)
4. `const rawStream = backend.readStream(path)`
5. Create a TransformStream that:
   - For csv/tsv: accumulates chunks, splits on newlines, parses each complete line with papaparse
   - For text: splits on newlines, emits each line as a string
6. Return rawStream.pipeThrough(transformStream)
